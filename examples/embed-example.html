Add this to your HTML to embed the game:

Basic Usage

// Import the library and default config
import { initializeGame, DEFAULT_GAME_CONFIG } from 'business-card-breakout';

// Initialize with default settings (recommended)
const engine = initializeGame(
  'game-container',  // Container ID
  {
    name: 'Your Name',
    title: 'Your Title',
    company: 'Your Company',
    email: 'your@email.com'
  },
  DEFAULT_GAME_CONFIG,  // Use default settings
  'standard',  // Layout: 'professional' | 'standard' | 'minimal'
  false  // autoStart (best practice: false)
);

// Explicitly start the game
engine.start();

Custom Game Settings

// Customize game behavior
const customConfig = {
  ballSpeed: 8,  // Ball speed (default: 10)
  ballRadius: 10,  // Ball size (default: 12)
  paddleSpeed: 10,  // Paddle movement speed (default: 12)
  paddleWidthRatio: 0.15,  // Paddle width ratio (default: 0.2)
  paddleHeight: 3,  // Paddle height (default: 4)
  blockRecoveryTime: 10000,  // Block recovery time in ms (default: 10000)
  effectDuration: 5000,  // Visual effect duration in ms (default: 5000)
  destructionRadius: 30,  // Area around ball that destroys blocks (default: 30)
  gravity: 0,  // Gravity (default: 0)
  friction: 1.0  // Ball friction (default: 1.0)
};

const engine = initializeGame(
  'game-container',
  { /* business card info */ },
  customConfig,  // Use custom settings
  'minimal',
  false
);

engine.start();

Layout Options

// Available layouts:

// 'professional' - Full layout with all fields including SNS
// Best for: Complete business cards with social media

// 'standard' - Standard layout with basic contact info
// Best for: Traditional business cards
// Displays: Name, Title, Company, Email, Phone, Website

// 'minimal' - Minimal layout with essential info only
// Best for: Clean, simple cards
// Displays: Name, Title, Company, Email
